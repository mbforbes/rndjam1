\begin{align*}
\r^{(t+1)} &= \y - \X\w^{(t+1)} \\
&= \y - \left( \X\w^{(t)} - w_j^{(t)} \X^j + w_j^{(t+1)} \X^j \right) \\
&= \y - \left( \X\w^{(t)} + (w_j^{(t+1)} - w_j^{(t)} ) \X^j \right) \\
&= \y - \X\w^{(t)} +(w_j^{(t)} - w_j^{(t+1)} ) \X^j \\
&= \r^{(t)} + (w_j^{(t)} - w_j^{(t+1)}) \X^j \\
\end{align*}
