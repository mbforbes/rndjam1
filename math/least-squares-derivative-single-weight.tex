\begin{align*}
\frac{\partial}{\partial w_j} \mathcal{L}_{LS}(\X,\y,\w) &= \frac{\partial}{\partial w_j} \left[ \frac{1}{n} \|\y - \X\w||_2^2 \right] \\
&= \frac{2}{n} (\y - \X\w)^T (-\X^j) \\
&= \frac{2}{n} (\X^j)^T (\X\w - \y) \\
\end{align*}
