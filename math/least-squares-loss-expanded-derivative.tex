% actual loss and derivative
\begin{align*}
\frac{\partial}{\partial \w} \mathcal{L}_{LS} (\X,\y,\w) &= \frac{\partial}{\partial \w} \left( \y^T\y - 2\y^T\X\w + \w^T\X^T\X\w \right) \\
&= -2\y^T\X + 2\w^T\X^T\X \\
\end{align*}
